    let ecoPoints = 0;
    let cart = [];
    let history = [];

    function login() {
      let email = document.getElementById("email").value;
      let pass = document.getElementById("password").value;
      if(email && pass){
        document.getElementById("loginPage").style.display = "none";
        document.getElementById("profilePage").style.display = "block";
        document.getElementById("nav").style.display = "block";
      } else {
        alert("Please enter email & password");
      }
    }

    function saveProfile(){
      let username = document.getElementById("username").value;
      if(username){
        alert("Profile saved for: " + username);
      } else {
        alert("Enter a username");
      }
    }

    function showProfile(){
      hideAll();
      document.getElementById("profilePage").style.display = "block";
    }

    function showProducts(){
      hideAll();
      document.getElementById("productsPage").style.display = "block";
    }

    function showCart(){
      hideAll();
      document.getElementById("cartPage").style.display = "block";
      renderCart();
    }

    function showHistory(){
      hideAll();
      document.getElementById("historyPage").style.display = "block";
      renderHistory();
    }

    function hideAll(){
      document.getElementById("profilePage").style.display = "none";
      document.getElementById("productsPage").style.display = "none";
      document.getElementById("cartPage").style.display = "none";
      document.getElementById("historyPage").style.display = "none";
    }

    function addToCart(item, price){
      cart.push({item, price});
      alert(item + " added to cart");
    }

    function renderCart(){
      let container = document.getElementById("cartItems");
      container.innerHTML = "";
      cart.forEach((c, i)=>{
        container.innerHTML += <p>${c.item} - â‚¹${c.price}</p>;
      });
    }

    function checkout(){
      history = history.concat(cart);
      cart = [];
      ecoPoints += 10; // reward eco points
      document.getElementById("ecoPoints").innerText = ecoPoints;
      alert("Checkout complete! You earned 10 EcoPoints ðŸŒ±");
      showHistory();
    }

    function renderHistory(){
      let container = document.getElementById("purchaseHistory");
      container.innerHTML = "";
      history.forEach(h =>{
        container.innerHTML += <p>${h.item} - â‚¹${h.price}</p>;
      });
    }
